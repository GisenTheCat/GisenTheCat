if (document.loadedCustomScript == true) throw null; else document.loadedCustomScript = true;
console.log('script loaded!')
if (document.location.href != 'https://classroom.google.com/h') throw null;
const classroomRenames = [
    'Español',
    'Teknik',
    'NO',
    'Hkk',
    'SO',
    'Output',
    'Musik',
    'Bild',
    'Idh',
    'English',
    'Ma',
    'Svenska',
    'Slöjd',
    'Mentorsklassrum'
] // if a classroom includes a word here, rename it to that
let whenClassroomsLoad = setInterval(() => {
    if (Array.from(document.getElementsByClassName('YVvGBb z3vRcc-ZoZQ1')).length == 0) return;
    clearInterval(whenClassroomsLoad);
    Array.from(document.getElementsByClassName('YVvGBb z3vRcc-ZoZQ1')).forEach((e) => {
        classroomRenames.forEach((name) => {
            if (e.innerText.toLowerCase().includes(name.toLowerCase())) {
                e.innerText = name;
            }
        });
    });
}, 0)
